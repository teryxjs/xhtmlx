Trigger syntax summary

<!-- Default: click for buttons -->
<button xh-get="/api/data">Click me</button>

<!-- Auto-load on page load -->
<div xh-get="/api/data" xh-trigger="load"></div>

<!-- Debounced keyup for search -->
<input xh-get="/api/search" xh-trigger="keyup changed delay:300ms">

<!-- Polling every 5 seconds -->
<div xh-get="/api/feed" xh-trigger="every 5s"></div>

<!-- Fire only once -->
<button xh-get="/api/data" xh-trigger="click once"></button>

<!-- Load when scrolled into view -->
<div xh-get="/api/more" xh-trigger="revealed"></div>

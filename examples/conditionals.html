Key syntax

<!-- Show only if field is truthy -->
<span xh-if="is_admin" class="badge">Admin</span>

<!-- Show only if field is falsy -->
<span xh-unless="verified" class="warning">Unverified</span>

<!-- Conditional block -->
<div xh-if="is_admin" class="admin-panel">
  Admin content here...
</div>

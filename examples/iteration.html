Key syntax

<!-- Simple iteration -->
<li xh-each="users">
  <span xh-text="name"></span>
</li>

<!-- With $index -->
<li xh-each="users">
  #<span xh-text="$index"></span>: <span xh-text="name"></span>
</li>

<!-- Nested: posts each have comments via separate API call -->
<div xh-each="posts">
  <h4 xh-text="title"></h4>
  <div xh-get="/api/posts/{{id}}/comments" xh-trigger="load">
    <template>
      <div xh-each="comments">...</div>
    </template>
  </div>
</div>

Key attributes used

     xh-trigger="load"             -- Fire on page load
     xh-template="/tpl/card.html"  -- External template
     xh-indicator="#spinner">      -- Loading indicator
  <span class="xh-indicator">Loading...</span>
</div>

<!-- In the template: -->
<span xh-text="name"></span>          -- Text binding
<img xh-attr-src="avatar">            -- Attribute binding
<span xh-if="is_admin">Admin</span>  -- Conditional
